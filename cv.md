function sortByBit(arr) {
  let numberInBits = (n) => n.toString(2).replace(new RegExp('0', 'g'), '');
  return arr.sort((a, b) => a - b).sort((x, y) =>
    numberInBits(x) - numberInBits(y) || x - y)
}
